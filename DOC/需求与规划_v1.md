## 核心需求

构建一个面向 YOLO 目标检测标签修正工作的 Web UI 工具，解决三类标签问题：
- **Overlooked（漏标）**：GT 中遗漏了应标注的目标
- **Swapped Class（类别错误）**：GT 中目标的类别标注错误  
- **Badly Located（框不准）**：GT 中边界框位置/尺寸偏差大

## 数据格式定义

### 1. YOLO 标签格式 (输入/输出)

**GT 标签格式**（无置信度）:
```
# 每行: class_id center_x center_y width height (归一化到0~1)
0 0.5 0.5 0.2 0.3
1 0.3 0.7 0.15 0.25
```

**Pred 标签格式**（ultralytics save_txt with save_conf=True）:
```
# 每行: class_id center_x center_y width height confidence
0 0.5 0.5 0.2 0.3 0.95
1 0.3 0.7 0.15 0.25 0.87
```

### 1.1 类别名称映射

支持两种类别名称配置文件：

**classes.txt 格式**:
```
person
car
bicycle
```

**data.yaml 格式** (ultralytics 风格):
```yaml
names:
  0: person
  1: car
  2: bicycle
# 或
names: ['person', 'car', 'bicycle']
```

### 2. Cleanlab 要求格式 (内部处理)

**Labels 格式**：
```python
labels = [
    {
        'bboxes': np.array([[x1, y1, x2, y2], ...]),  # 像素坐标, shape=(L, 4)
        'labels': np.array([class_id, ...]),          # shape=(L,)
        'image_name': 'image_001.jpg'                 # 可选
    },
    ...
]
```

**Predictions 格式**：
```python
predictions = [
    # 第i张图片的预测: 按类别组织的数组
    np.array([
        # 类别0的预测框: shape=(M0, 5), [x1, y1, x2, y2, confidence]
        np.array([[x1, y1, x2, y2, conf], ...]),
        # 类别1的预测框: shape=(M1, 5)
        np.array([[x1, y1, x2, y2, conf], ...]),
        ...
    ], dtype=object),
    ...
]
```

### 3. UI 内部格式 (标注组件)
```python
class BBox:
    x: float      # 左上角 x (像素)
    y: float      # 左上角 y (像素)  
    w: float      # 宽度 (像素)
    h: float      # 高度 (像素)
    class_id: int # 类别ID
    source: str   # 'gt' | 'pred'
```

```python
from dataclasses import dataclass, field
from typing import List, Optional, Dict
from pathlib import Path
from enum import Enum
import uuid

class IssueType(Enum):
    OVERLOOKED = "overlooked"
    SWAPPED = "swapped"
    BAD_LOCATED = "bad_located"

class BoxSource(Enum):
    GT = "gt"
    PRED = "pred"

@dataclass
class BBox:
    """UI层使用的边界框"""
    x: float                    # 左上角x (像素)
    y: float                    # 左上角y (像素)
    w: float                    # 宽度 (像素)
    h: float                    # 高度 (像素)
    class_id: int               # 类别ID
    source: BoxSource           # 来源
    id: str = field(default_factory=lambda: str(uuid.uuid4()))
    selected: bool = False

@dataclass
class IssueItem:
    """问题样本"""
    image_name: str             # 图片文件名
    issue_type: IssueType       # 问题类型
    score: float                # 问题严重程度分数 (越低越严重)
    box_index: Optional[int]    # 问题框索引 (用于高亮)

@dataclass  
class SessionConfig:
    """会话配置"""
    images_path: str = ""
    pred_labels_path: str = ""
    gt_labels_path: str = ""
    classes_file: str = ""      # 可选：classes.txt 或 data.yaml 路径
    top_k: int = 10
    backup_enabled: bool = True

@dataclass
class ClassMapping:
    """类别映射"""
    id_to_name: Dict[int, str]  # class_id -> class_name
    
    @classmethod
    def from_file(cls, file_path: str) -> 'ClassMapping':
        """从 classes.txt 或 data.yaml 加载类别映射"""
        path = Path(file_path)
        id_to_name = {}
        
        if path.suffix == '.txt':
            with open(path, 'r') as f:
                for i, line in enumerate(f):
                    name = line.strip()
                    if name:
                        id_to_name[i] = name
        elif path.suffix in ('.yaml', '.yml'):
            import yaml
            with open(path, 'r') as f:
                data = yaml.safe_load(f)
            names = data.get('names', {})
            if isinstance(names, list):
                id_to_name = {i: n for i, n in enumerate(names)}
            elif isinstance(names, dict):
                id_to_name = {int(k): v for k, v in names.items()}
        
        return cls(id_to_name=id_to_name)
    
    def get_name(self, class_id: int) -> str:
        """获取类别名称，若无则返回 str(class_id)"""
        return self.id_to_name.get(class_id, str(class_id))
    
    def get_display(self, class_id: int) -> str:
        """获取显示字符串: 'id: name' 或 'id'"""
        if class_id in self.id_to_name:
            return f"{class_id}: {self.id_to_name[class_id]}"
        return str(class_id)
```

### 2. `src/core/yolo_utils.py` - 格式转换

```python
import numpy as np
from pathlib import Path
from typing import List, Tuple, Dict, Any
from PIL import Image

def yolo_to_pixel(cx: float, cy: float, w: float, h: float, 
                  img_w: int, img_h: int) -> Tuple[float, float, float, float]:
    """YOLO归一化坐标 -> 像素坐标 (x1, y1, x2, y2)"""
    box_w = w * img_w
    box_h = h * img_h
    x1 = (cx * img_w) - (box_w / 2)
    y1 = (cy * img_h) - (box_h / 2)
    x2 = x1 + box_w
    y2 = y1 + box_h
    return x1, y1, x2, y2

def pixel_to_yolo(x1: float, y1: float, x2: float, y2: float,
                  img_w: int, img_h: int) -> Tuple[float, float, float, float]:
    """像素坐标 (x1, y1, x2, y2) -> YOLO归一化坐标"""
    cx = ((x1 + x2) / 2) / img_w
    cy = ((y1 + y2) / 2) / img_h
    w = (x2 - x1) / img_w
    h = (y2 - y1) / img_h
    return cx, cy, w, h

def read_yolo_label(label_path: Path, img_w: int, img_h: int) -> List[Dict]:
    """读取YOLO标签文件，返回像素坐标格式"""
    boxes = []
    if not label_path.exists():
        return boxes
    with open(label_path, 'r') as f:
        for line in f:
            parts = line.strip().split()
            if len(parts) >= 5:
                class_id = int(parts[0])
                cx, cy, w, h = map(float, parts[1:5])
                x1, y1, x2, y2 = yolo_to_pixel(cx, cy, w, h, img_w, img_h)
                boxes.append({
                    'class_id': class_id,
                    'bbox': [x1, y1, x2, y2]
                })
    return boxes

def write_yolo_label(label_path: Path, boxes: List[Dict], img_w: int, img_h: int):
    """将像素坐标格式的框写入YOLO标签文件"""
    with open(label_path, 'w') as f:
        for box in boxes:
            x1, y1, x2, y2 = box['bbox']
            cx, cy, w, h = pixel_to_yolo(x1, y1, x2, y2, img_w, img_h)
            f.write(f"{box['class_id']} {cx:.6f} {cy:.6f} {w:.6f} {h:.6f}\n")

def prepare_cleanlab_format(images_dir: Path, labels_dir: Path) -> Tuple[List[Dict], List[str]]:
    """
    将YOLO格式数据转换为Cleanlab所需格式
    
    Returns:
        labels: cleanlab格式的标签列表
        image_names: 图片文件名列表 (保持顺序对应)
    """
    labels = []
    image_names = []
    
    image_extensions = {'.jpg', '.jpeg', '.png', '.bmp'}
    image_files = sorted([f for f in images_dir.iterdir() 
                          if f.suffix.lower() in image_extensions])
    
    for img_path in image_files:
        # 获取图片尺寸
        with Image.open(img_path) as img:
            img_w, img_h = img.size
        
        # 查找对应标签文件
        label_path = labels_dir / f"{img_path.stem}.txt"
        boxes = read_yolo_label(label_path, img_w, img_h)
        
        if boxes:
            bboxes = np.array([b['bbox'] for b in boxes])
            class_ids = np.array([b['class_id'] for b in boxes])
        else:
            bboxes = np.zeros((0, 4))
            class_ids = np.array([], dtype=int)
        
        labels.append({
            'bboxes': bboxes,
            'labels': class_ids,
            'image_name': img_path.name
        })
        image_names.append(img_path.name)
    
    return labels, image_names
```

### 3. `src/core/analyzer.py` - Cleanlab 分析器

```python
from typing import List, Dict, Callable, Optional
from pathlib import Path
import numpy as np

from cleanlab.object_detection.filter import find_label_issues
from cleanlab.object_detection.rank import (
    get_label_quality_scores,
    compute_overlooked_box_scores,
    compute_swap_box_scores,
    compute_badloc_box_scores
)

from ..models import IssueItem, IssueType
from .yolo_utils import prepare_cleanlab_format

class CleanlabAnalyzer:
    """Cleanlab 目标检测分析器"""
    
    def __init__(self, 
                 images_path: str,
                 pred_labels_path: str,
                 gt_labels_path: str,
                 progress_callback: Optional[Callable[[str, float], None]] = None):
        self.images_path = Path(images_path)
        self.pred_labels_path = Path(pred_labels_path)
        self.gt_labels_path = Path(gt_labels_path)
        self.progress_callback = progress_callback or (lambda msg, pct: None)
        
        self.labels = []        # cleanlab格式的GT
        self.predictions = []   # cleanlab格式的预测
        self.image_names = []   # 图片文件名列表
    
    def _report_progress(self, message: str, percentage: float):
        """报告进度"""
        self.progress_callback(message, percentage)
    
    def prepare_data(self):
        """准备数据：格式转换"""
        self._report_progress("Converting GT labels...", 0.1)
        self.labels, self.image_names = prepare_cleanlab_format(
            self.images_path, self.gt_labels_path
        )
        
        self._report_progress("Converting Pred labels...", 0.3)
        self.predictions = self._convert_predictions()
        
        self._report_progress("Data preparation complete", 0.4)
    
    def _convert_predictions(self) -> List[np.ndarray]:
        """将YOLO预测转换为Cleanlab格式"""
        from PIL import Image
        
        predictions = []
        # 需要确定类别数量
        all_classes = set()
        for label in self.labels:
            all_classes.update(label['labels'].tolist())
        # 从预测中也收集
        for img_name in self.image_names:
            pred_path = self.pred_labels_path / f"{Path(img_name).stem}.txt"
            if pred_path.exists():
                with open(pred_path, 'r') as f:
                    for line in f:
                        parts = line.strip().split()
                        if len(parts) >= 5:
                            all_classes.add(int(parts[0]))
        
        num_classes = max(all_classes) + 1 if all_classes else 1
        
        for img_name in self.image_names:
            img_path = self.images_path / img_name
            with Image.open(img_path) as img:
                img_w, img_h = img.size
            
            # 按类别组织预测
            pred_by_class = [[] for _ in range(num_classes)]
            
            pred_path = self.pred_labels_path / f"{Path(img_name).stem}.txt"
            if pred_path.exists():
                with open(pred_path, 'r') as f:
                    for line in f:
                        parts = line.strip().split()
                        if len(parts) >= 6:  # class cx cy w h conf
                            class_id = int(parts[0])
                            cx, cy, w, h = map(float, parts[1:5])
                            conf = float(parts[5]) if len(parts) > 5 else 1.0
                            
                            # 转换坐标
                            from .yolo_utils import yolo_to_pixel
                            x1, y1, x2, y2 = yolo_to_pixel(cx, cy, w, h, img_w, img_h)
                            
                            if class_id < num_classes:
                                pred_by_class[class_id].append([x1, y1, x2, y2, conf])
            
            # 转换为numpy数组
            pred_array = np.array([
                np.array(boxes) if boxes else np.zeros((0, 5))
                for boxes in pred_by_class
            ], dtype=object)
            
            predictions.append(pred_array)
        
        return predictions
    
    def analyze(self, top_k: int = 10) -> Dict[IssueType, List[IssueItem]]:
        """
        执行Cleanlab分析
        
        Returns:
            按问题类型分组的IssueItem列表
        """
        self._report_progress("Running Cleanlab analysis...", 0.5)
        
        # 计算各类分数
        overlooked_scores = compute_overlooked_box_scores(
            labels=self.labels, predictions=self.predictions
        )
        
        self._report_progress("Computing swap scores...", 0.6)
        swap_scores = compute_swap_box_scores(
            labels=self.labels, predictions=self.predictions
        )
        
        self._report_progress("Computing bad location scores...", 0.7)
        badloc_scores = compute_badloc_box_scores(
            labels=self.labels, predictions=self.predictions
        )
        
        self._report_progress("Ranking issues...", 0.8)
        
        results = {
            IssueType.OVERLOOKED: [],
            IssueType.SWAPPED: [],
            IssueType.BAD_LOCATED: []
        }
        
        # 对于 overlooked: 按预测框分数排序 (低分 = 更可能漏标)
        overlooked_items = []
        for i, (img_name, scores) in enumerate(zip(self.image_names, overlooked_scores)):
            if len(scores) > 0:
                min_score = float(np.min(scores))
                min_idx = int(np.argmin(scores))
                overlooked_items.append(IssueItem(
                    image_name=img_name,
                    issue_type=IssueType.OVERLOOKED,
                    score=min_score,
                    box_index=min_idx
                ))
        overlooked_items.sort(key=lambda x: x.score)
        results[IssueType.OVERLOOKED] = overlooked_items[:top_k]
        
        # 对于 swapped: 按GT框分数排序
        swapped_items = []
        for i, (img_name, scores) in enumerate(zip(self.image_names, swap_scores)):
            if len(scores) > 0:
                min_score = float(np.min(scores))
                min_idx = int(np.argmin(scores))
                swapped_items.append(IssueItem(
                    image_name=img_name,
                    issue_type=IssueType.SWAPPED,
                    score=min_score,
                    box_index=min_idx
                ))
        swapped_items.sort(key=lambda x: x.score)
        results[IssueType.SWAPPED] = swapped_items[:top_k]
        
        # 对于 bad_located: 按GT框分数排序
        badloc_items = []
        for i, (img_name, scores) in enumerate(zip(self.image_names, badloc_scores)):
            if len(scores) > 0:
                min_score = float(np.min(scores))
                min_idx = int(np.argmin(scores))
                badloc_items.append(IssueItem(
                    image_name=img_name,
                    issue_type=IssueType.BAD_LOCATED,
                    score=min_score,
                    box_index=min_idx
                ))
        badloc_items.sort(key=lambda x: x.score)
        results[IssueType.BAD_LOCATED] = badloc_items[:top_k]
        
        self._report_progress("Analysis complete", 1.0)
        return results
```

from pathlib import Path
from datetime import datetime
import shutil
from typing import List

def backup_folder(folder_path: str) -> str:
    """
    备份文件夹
    
    Returns:
        备份目录路径
    """
    src = Path(folder_path)
    if not src.exists():
        raise FileNotFoundError(f"Source folder not found: {folder_path}")
    
    # 生成备份目录名
    backup_name = f"{src.name}_bk"
    backup_path = src.parent / backup_name
    
    # 若已存在，添加时间戳
    if backup_path.exists():
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_name = f"{src.name}_bk_{timestamp}"
        backup_path = src.parent / backup_name
    
    shutil.copytree(src, backup_path)
    return str(backup_path)

def save_tmp_annotation(gt_labels_path: str, image_name: str, 
                        boxes: List[dict], img_w: int, img_h: int):
    """保存临时标注文件"""
    from .yolo_utils import write_yolo_label
    
    gt_dir = Path(gt_labels_path)
    stem = Path(image_name).stem
    tmp_path = gt_dir / f"{stem}_tmp.txt"
    
    write_yolo_label(tmp_path, boxes, img_w, img_h)

def confirm_changes(gt_labels_path: str, keep_changes: bool):
    """
    确认或放弃更改
    
    Args:
        gt_labels_path: GT标签目录
        keep_changes: True=用tmp覆盖原文件, False=删除tmp
    """
    gt_dir = Path(gt_labels_path)
    tmp_files = list(gt_dir.glob("*_tmp.txt"))
    
    for tmp_file in tmp_files:
        # 从 xxx_tmp.txt 推断原文件名 xxx.txt
        original_name = tmp_file.name.replace("_tmp.txt", ".txt")
        original_path = gt_dir / original_name
        
        if keep_changes:
            # 覆盖原文件
            shutil.move(str(tmp_file), str(original_path))
        else:
            # 删除临时文件
            tmp_file.unlink()

def get_tmp_files(gt_labels_path: str) -> List[str]:
    """获取所有临时文件列表"""
    gt_dir = Path(gt_labels_path)
    return [f.name for f in gt_dir.glob("*_tmp.txt")]

### 5. `src/ui/components.py` - 标注组件

```python
from nicegui import ui, events
from typing import List, Optional, Callable
from dataclasses import dataclass
import uuid

@dataclass
class AnnotationBox:
    """标注框"""
    x: float
    y: float
    w: float
    h: float
    class_id: int
    source: str  # 'gt' or 'pred'
    id: str = None
    
    def __post_init__(self):
        if self.id is None:
            self.id = str(uuid.uuid4())

class InteractiveAnnotator:
    """交互式标注组件"""
    
    # 颜色配置
    COLORS = {
        'gt': {'normal': '#22c55e', 'selected': '#ef4444'},      # 绿色 / 红色
        'pred': {'normal': '#3b82f6', 'selected': '#ef4444'},    # 蓝色 / 红色
    }
    
    MAX_HISTORY = 50  # 最大撤销步数
    
    def __init__(self, 
                 on_change: Callable[[List[AnnotationBox]], None] = None,
                 show_border: bool = True,
                 show_fill: bool = False):
        
        self.gt_boxes: List[AnnotationBox] = []
        self.pred_boxes: List[AnnotationBox] = []
        self.selected_box: Optional[AnnotationBox] = None
        self.on_change = on_change or (lambda x: None)
        
        # 显示选项
        self.show_gt = True
        self.show_pred = True
        self.show_border = show_border
        self.show_fill = show_fill
        
        # 交互状态
        self.mode = "IDLE"  # IDLE, DRAWING, MOVING, RESIZING
        self.start_x = 0
        self.start_y = 0
        self.resize_handle = None  # 'nw', 'ne', 'sw', 'se', 'n', 's', 'e', 'w'
        self.drawing_box: Optional[AnnotationBox] = None
        
        # 撤销/重做栈
        self.undo_stack: List[List[AnnotationBox]] = []  # 历史状态栈
        self.redo_stack: List[List[AnnotationBox]] = []  # 重做状态栈
        
        # 图片信息
        self.img_width = 0
        self.img_height = 0
        self.current_class_id = 0
        
        # UI 组件
        self.image_component = None
    
    def create_ui(self, container):
        """在指定容器中创建UI"""
        with container:
            self.image_component = ui.interactive_image(
                '',
                on_mouse=self._handle_mouse,
                events=['mousedown', 'mouseup', 'mousemove'],
                cross=True,
                sanitize=False
            ).classes('w-full border rounded')
    
    def load_image(self, image_path: str, img_width: int, img_height: int):
        """加载图片"""
        self.img_width = img_width
        self.img_height = img_height
        self.image_component.set_source(image_path)
        self._update_svg()
    
    def load_boxes(self, gt_boxes: List[AnnotationBox], pred_boxes: List[AnnotationBox]):
        """加载标注框"""
        self.gt_boxes = gt_boxes
        self.pred_boxes = pred_boxes
        self.selected_box = None
        self._update_svg()
    
    def _box_to_svg(self, box: AnnotationBox) -> str:
        """将框转换为SVG字符串"""
        is_selected = (self.selected_box and self.selected_box.id == box.id)
        color = self.COLORS[box.source]['selected' if is_selected else 'normal']
        
        stroke_width = 3 if (self.show_border or is_selected) else 1
        fill_opacity = 0.2 if (self.show_fill or is_selected) else 0
        
        # 添加class标签
        label_y = max(box.y - 5, 12)
        
        return f'''
            <rect x="{box.x}" y="{box.y}" width="{box.w}" height="{box.h}"
                  stroke="{color}" stroke-width="{stroke_width}" 
                  fill="{color}" fill-opacity="{fill_opacity}"
                  data-id="{box.id}" pointer-events="all" cursor="pointer"/>
            <text x="{box.x}" y="{label_y}" fill="{color}" font-size="12">
                {box.class_id}
            </text>
        '''
    
    def _update_svg(self):
        """更新SVG内容"""
        if not self.image_component:
            return
        
        svg_parts = []
        
        # 渲染 Pred 框
        if self.show_pred:
            for box in self.pred_boxes:
                svg_parts.append(self._box_to_svg(box))
        
        # 渲染 GT 框
        if self.show_gt:
            for box in self.gt_boxes:
                svg_parts.append(self._box_to_svg(box))
        
        # 渲染正在绘制的框
        if self.drawing_box:
            svg_parts.append(self._box_to_svg(self.drawing_box))
        
        self.image_component.set_content(''.join(svg_parts))
    
    def _find_box_at(self, x: float, y: float) -> Optional[AnnotationBox]:
        """查找指定位置的框（优先GT，倒序遍历）"""
        if self.show_gt:
            for box in reversed(self.gt_boxes):
                if box.x <= x <= box.x + box.w and box.y <= y <= box.y + box.h:
                    return box
        return None
    
    def _handle_mouse(self, e: events.MouseEventArguments):
        """处理鼠标事件"""
        x, y = e.image_x, e.image_y
        
        if e.type == 'mousedown':
            clicked_box = self._find_box_at(x, y)
            
            if clicked_box:
                # 选中已有框，准备移动
                self._save_state()  # 移动前保存状态
                self.selected_box = clicked_box
                self.mode = "MOVING"
                self.start_x, self.start_y = x, y
            else:
                # 开始画新框
                self._save_state()  # 绘制前保存状态
                self.selected_box = None
                self.mode = "DRAWING"
                self.start_x, self.start_y = x, y
                self.drawing_box = AnnotationBox(
                    x=x, y=y, w=0, h=0,
                    class_id=self.current_class_id,
                    source='gt'
                )
        
        elif e.type == 'mousemove':
            if self.mode == "DRAWING" and self.drawing_box:
                # 更新绘制中的框
                self.drawing_box.w = x - self.start_x
                self.drawing_box.h = y - self.start_y
            
            elif self.mode == "MOVING" and self.selected_box:
                # 移动选中的框
                dx = x - self.start_x
                dy = y - self.start_y
                self.selected_box.x += dx
                self.selected_box.y += dy
                self.start_x, self.start_y = x, y
        
        elif e.type == 'mouseup':
            if self.mode == "DRAWING" and self.drawing_box:
                # 规范化框（处理负宽高）
                if self.drawing_box.w < 0:
                    self.drawing_box.x += self.drawing_box.w
                    self.drawing_box.w = abs(self.drawing_box.w)
                if self.drawing_box.h < 0:
                    self.drawing_box.y += self.drawing_box.h
                    self.drawing_box.h = abs(self.drawing_box.h)
                
                # 过滤太小的框
                if self.drawing_box.w >= 5 and self.drawing_box.h >= 5:
                    self.gt_boxes.append(self.drawing_box)
                    self.on_change(self.gt_boxes)
                else:
                    # 无效绘制，撤销保存的状态
                    if self.undo_stack:
                        self.undo_stack.pop()
                
                self.drawing_box = None
            
            elif self.mode == "MOVING":
                self.on_change(self.gt_boxes)
            
            self.mode = "IDLE"
        
        self._update_svg()
    
    def _save_state(self):
        """保存当前状态到撤销栈"""
        import copy
        state = copy.deepcopy(self.gt_boxes)
        self.undo_stack.append(state)
        # 限制栈大小
        if len(self.undo_stack) > self.MAX_HISTORY:
            self.undo_stack.pop(0)
        # 新操作清空重做栈
        self.redo_stack.clear()
    
    def undo(self):
        """撤销上一步操作"""
        if not self.undo_stack:
            return False
        import copy
        # 保存当前状态到重做栈
        self.redo_stack.append(copy.deepcopy(self.gt_boxes))
        # 恢复上一个状态
        self.gt_boxes = self.undo_stack.pop()
        self.selected_box = None
        self._update_svg()
        self.on_change(self.gt_boxes)
        return True
    
    def redo(self):
        """重做已撤销的操作"""
        if not self.redo_stack:
            return False
        import copy
        # 保存当前状态到撤销栈
        self.undo_stack.append(copy.deepcopy(self.gt_boxes))
        # 恢复重做状态
        self.gt_boxes = self.redo_stack.pop()
        self.selected_box = None
        self._update_svg()
        self.on_change(self.gt_boxes)
        return True
    
    def delete_selected(self):
        """删除选中的框"""
        if self.selected_box and self.selected_box in self.gt_boxes:
            self._save_state()  # 保存状态用于撤销
            self.gt_boxes.remove(self.selected_box)
            self.selected_box = None
            self._update_svg()
            self.on_change(self.gt_boxes)
    
    def set_display_options(self, show_gt: bool = None, show_pred: bool = None,
                           show_border: bool = None, show_fill: bool = None):
        """设置显示选项"""
        if show_gt is not None:
            self.show_gt = show_gt
        if show_pred is not None:
            self.show_pred = show_pred
        if show_border is not None:
            self.show_border = show_border
        if show_fill is not None:
            self.show_fill = show_fill
        self._update_svg()
```

## 关键技术点说明

### 1. Cleanlab API 调用注意事项

Cleanlab 的 Object Detection 模块要求：
- **坐标格式**: 必须使用 `[x1, y1, x2, y2]` 像素坐标
- **预测格式**: 必须按类别组织，每类一个 `(M, 5)` 数组
- **置信度**: 预测需包含 confidence 列

原 YOLO 格式使用归一化的 `(cx, cy, w, h)`，需要严格转换。

### 2. NiceGUI interactive_image 使用要点

```python
# 关键参数
ui.interactive_image(
    source,                    # 图片路径或URL
    on_mouse=handler,          # 鼠标事件回调
    events=['mousedown', ...], # 监听的事件类型
    cross=True,                # 显示十字准星
    sanitize=False             # 允许自定义SVG
)

# 事件对象属性
e.type       # 'mousedown', 'mouseup', 'mousemove'
e.image_x    # 相对于图片的X坐标 (像素)
e.image_y    # 相对于图片的Y坐标 (像素)
```

### 3. 临时文件命名规则

```
原GT文件:    gt_labels/image_001.txt
临时文件:    gt_labels/image_001_tmp.txt
确认后:      gt_labels/image_001.txt (覆盖)
```
